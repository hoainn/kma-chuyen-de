\section{Phụ lục: Tái lập kết quả}
\label{app:repro}

Phụ lục cho phép một người đánh giá độc lập \emph{chạy lại} thực nghiệm ngoại tuyến.
\emph{Mọi công cụ chạy bằng Docker}---không cần cài đặt trực tiếp lên máy chủ ngoài
Docker.

\subsection{Chuẩn bị dữ liệu DongTing đầy đủ}
\begin{lstlisting}[language=bash]
# DongTing trong kho chi la MAU. Tai ban DAY DU (Normal_data/, Abnormal_data/,
# Baseline.xlsx, syscall_64.tbl) va tro DATA_ROOT toi thu muc giai nen.
export DATA_ROOT=/duong/dan/toi/dongting-full
\end{lstlisting}

\subsection{Huấn luyện ngoại tuyến (sinh số cho Bảng~\ref{tab:offline})}
\begin{lstlisting}[language=bash]
cd DeSFAM/training
DATA_ROOT=$DATA_ROOT docker compose -f docker-compose.pipeline.yml run --rm train
# Quan sat chi so in ra; ghi lai ket qua + nhan xet vao Muc 7.
\end{lstlisting}

\subsection{Kiểm tra tính nhất quán đặc trưng (train $\leftrightarrow$ serve)}
\begin{lstlisting}[language=bash]
cd DeSFAM
docker compose -f training/docker-compose.pipeline.yml run --rm \
  train python -m pytest tests/test_feature_parity.py -q
\end{lstlisting}

\subsection{Biên dịch báo cáo này (Docker)}
\begin{lstlisting}[language=bash]
cd ReportV2
./build.sh        # chay XeLaTeX qua Docker texlive -> main.pdf
\end{lstlisting}
